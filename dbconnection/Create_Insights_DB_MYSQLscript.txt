create table insights_db.environment (
systemID int not null auto_increment,
systemName varchar(45) unique not null,

CREATE TABLE `insights_db`.`user_customer` (
  `userName` VARCHAR(45) NOT NULL,
  `customerName` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`userName`, `customerName`),
  INDEX `userName_idx` (`customerName` ASC),
  CONSTRAINT `userName`
    FOREIGN KEY (`userName`)
    REFERENCES `insights_db`.`user` (`userName`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `customerName`
    FOREIGN KEY (`customerName`)
    REFERENCES `insights_db`.`customer` (`customerName`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
